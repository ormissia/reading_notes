```dataviewjs
let pages = dv.pages('"微信读书"')

const groupData = pages.groupBy(b => { let date = dv.date(b.lastReadDate); return `${date.year}-${String(date.month).padStart(2, '0')}`});
let labels = groupData.map(gd => gd.key).values;
let values = groupData.map(gd => gd.rows.length).values;

dv.span(`一共阅读了 ${groupData.rows.length} 本书！<br>`);
console.log(dv.func.sum(groupData.rows.readingtimestamp));
let totalReadingTime = dv.func.sum(groupData.rows.readingtimestamp);
let totalHours = Math.floor(totalReadingTime / 3600);
let totalMinutes = Math.floor((totalReadingTime % 3600) / 60);
dv.span(`一共阅读了 ${totalHours} 小时 ${totalMinutes} 分钟！<br>`);

const chartData = {
    type: 'bar',
    data: {
        labels: labels,
        datasets: [{
            label: '读书数量',
            data: values,
            backgroundColor: [
                'rgba(255, 99, 132, 0.2)'
            ],
            borderColor: [
                'rgba(255, 99, 132, 1)'
            ],
            borderWidth: 2
        }]
    }
}

window.renderChart(chartData, this.container);
```
